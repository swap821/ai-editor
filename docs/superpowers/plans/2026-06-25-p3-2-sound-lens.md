**Goal:** Add unit tests to `GAG demo/gag-orchestrator/src/lib/soundEngine.test.ts` that lock the existing sound-engine behavior, using a mock `AudioContext` and two test-only hooks exported from `soundEngine.ts`.

**Architecture:** Keep `soundEngine.ts` as a side-effect module. Expose `__resetSoundEngineForTests()` and `__setAudioContextForTests(ctx, master, ambGain)` so tests can inject a stub context and inspect the generated audio graph. The test file builds one reusable `createMockAudioContext()` factory that records every created node, connection, and scheduled parameter event.

**Tech Stack:** TypeScript, Vitest, Web Audio API mocks.

```typescript
import { vi } from 'vitest';

export interface MockAudioContext {
  state: AudioContextState;
  currentTime: number;
  sampleRate: number;
  destination: MockAudioNode;
  scheduled: ScheduledEvent[];
  nodes: MockAudioNode[];
  resume: ReturnType<typeof vi.fn>;
  close: ReturnType<typeof vi.fn>;
  createOscillator: () => MockAudioNode;
  createGain: () => MockAudioNode;
  createDynamicsCompressor: () => MockAudioNode;
  createBiquadFilter: () => MockAudioNode;
  createBuffer: (channels: number, length: number, sampleRate: number) => AudioBuffer;
  createBufferSource: () => MockAudioNode;
}

interface ScheduledEvent {
  node: MockAudioNode;
  param: string;
  method: string;
  args: unknown[];
}

interface MockAudioNode {
  type: string;
  frequency?: MockAudioParam;
  gain?: MockAudioParam;
  Q?: MockAudioParam;
  buffer: AudioBuffer | null;
  onstatechange: (() => void) | null;
  connect: ReturnType<typeof vi.fn>;
  disconnect: ReturnType<typeof vi.fn>;
  start: ReturnType<typeof vi.fn>;
  stop: ReturnType<typeof vi.fn>;
}

interface MockAudioParam {
  value: number;
  setValueAtTime: ReturnType<typeof vi.fn>;
  linearRampToValueAtTime: ReturnType<typeof vi.fn>;
  exponentialRampToValueAtTime: ReturnType<typeof vi.fn>;
  setTargetAtTime: ReturnType<typeof vi.fn>;
}

function createMockParam(node: MockAudioNode, scheduled: ScheduledEvent[]): MockAudioParam {
  const param: MockAudioParam = {
    value: 0,
    setValueAtTime: vi.fn((value, time) => scheduled.push({ node, param: 'value', method: 'setValueAtTime', args: [value, time] })),
    linearRampToValueAtTime: vi.fn((value, time) => scheduled.push({ node, param: 'value', method: 'linearRampToValueAtTime', args: [value, time] })),
    exponentialRampToValueAtTime: vi.fn((value, time) => scheduled.push({ node, param: 'value', method: 'exponentialRampToValueAtTime', args: [value, time] })),
    setTargetAtTime: vi.fn((value, time, timeConstant) => scheduled.push({ node, param: 'value', method: 'setTargetAtTime', args: [value, time, timeConstant] })),
  };
  return param;
}

export function createMockAudioContext(): MockAudioContext {
  const scheduled: ScheduledEvent[] = [];
  const nodes: MockAudioNode[] = [];

  function node(type: string): MockAudioNode {
    const n: MockAudioNode = {
      type,
      frequency: type === 'oscillator' || type === 'biquad' ? createMockParam(n as unknown as MockAudioNode, scheduled) : undefined,
      gain: type === 'gain' || type === 'compressor' ? createMockParam(n as unknown as MockAudioNode, scheduled) : undefined,
      Q: type === 'biquad' ? createMockParam(n as unknown as MockAudioNode, scheduled) : undefined,
      buffer: null,
      onstatechange: null,
      connect: vi.fn((dest) => dest),
      disconnect: vi.fn(),
      start: vi.fn(),
      stop: vi.fn(),
    };
    nodes.push(n);
    return n;
  }

  const destination = node('destination');

  return {
    state: 'running',
    currentTime: 1.0,
    sampleRate: 48000,
    destination,
    scheduled,
    nodes,
    resume: vi.fn(),
    close: vi.fn().mockResolvedValue(undefined),
    createOscillator: () => node('oscillator'),
    createGain: () => node('gain'),
    createDynamicsCompressor: () => node('compressor'),
    createBiquadFilter: () => node('biquad'),
    createBuffer: () => ({ getChannelData: () => new Float32Array(100) } as unknown as AudioBuffer),
    createBufferSource: () => node('bufferSource'),
  };
}
```

    __resetSoundEngineForTests();

    const oscs = mock.nodes.filter((n) => n.type === 'oscillator');

    for (const osc of ambientOscs) {
      const stopCall = osc.stop.mock.calls[0] as [number];
      expect(stopCall[0]).toBeGreaterThan(mock.currentTime);
    }

    vi.advanceTimersByTime(400);
